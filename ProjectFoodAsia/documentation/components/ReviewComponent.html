generateTxn(ActNo: number){
      
    let n = Math.round(Math.random()*100000000); 
    // console.log(n);
    let ns = n.toString();

    // console.log(ns.length);
      if(ns.length > 6){
          ns = ns.substr(0,6);
        //   console.log(ns);
      }

    let A = ActNo.toString();
    let txn = A + ns;

return txn
}
